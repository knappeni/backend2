function loadDoc(ddvalue) {
//hämta värdet från dropdown och spara det i en JS var
  var xhttp = new XMLHttpRequest();
 
  xhttp.onreadystatechange = function() {
    if (this.readyState == 4 && this.status == 200) {
     document.getElementById("ajaxfetch").innerHTML = this.responseText;
    }
  };
  xhttp.open("GET", "ajaxsok.php?query="+ddvalue, "true");
  xhttp.send();
}
